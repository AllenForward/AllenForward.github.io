JavaScript之模拟call和apply

发表于 2018-11-11 | 分类于 -JavaScript基础

call

在MDN中对call的定义如下：

call() 方法调用一个函数, 其具有一个指定的this值和分别地提供的参数(参数的列表)。

看个例子：

let obj = {
  a: 1
};
function say() {
  console.log(this.a);
}
say.call(obj); // 1

执行代码，我们发现call方法的作用大致有两点：

  1. 改变了this指向，this指向了obj
  2. 执行了say方法

我们可以简单地理解为：

obj.fn = say;
obj.fn();

但是这样的话我们就给obj新增了一个fn属性，所以我们在执行完obj.fn方法后，需要进行删除操作。

delete obj.fn;

这样，我们就实现call方法最简单的功能，但是这样写，并不能拿去用啊，好吧，我们下面就正式一点啦。

第一步

Function.prototype.myCall = function(context) {
  // 通过this获取调用call的函数
  context.fn = this;
  context.fn();
  delete context.fn;
}

// 测试一波
let obj = {
  a: 1
};
function say() {
  console.log(this.a);
}
say.myCall(obj); // 1

输出也为1，第一步模拟成功。

第二步

我们知道，call方法还可以给定参数执行函数。例如:

let obj = {
  a: 1
};
function say(name, age) {
  console.log(name);
  console.log(age);
  console.log(this.a);
}
say.call(obj, 'Allen', 24);
// Allen
// 24
// 1

因为传入的参数长度不固定，我们可以定义一个数组变量，将Arguments对象中第二个到最后一个参数放入其中，然后把这个数组变量传到要执行的函数参数里面。

let args = [];
for (let i = 1; i < arguments.length; i++) {
  args.push('arguments[' + i + ']');
}
// args: ['arguments[1]', 'arguments[2]', 'arguments[3]'......]
context.fn(args.join(','));

嗯，看起来也很简单嘛，但是等你运行代码，明显会一张黑人问号脸了。问题当然是出在把数组变量传到执行函数里面啦，因为我们这时传进函数的只是字符串。怎么办，我们可以使用eval方法。

eval('context.fn(' + args + ')');

国际惯例，我们整理一下第二版:

Function.prototype.myCall = function(context) {
  context.fn = this;
  let args = [];
  for (let i = 1; i < arguments.length; i++) {
    args.push('arguments[' + i + ']');
  }
  eval('context.fn('+ args +')');
  delete context.fn;
}
say.myCall(obj, 'Allen', 24);
// Allen
// 24
// 1

Nice啊！

第三步

模拟效果已经差不多达到80%了，但我们还需要注意三个点。

  1. this参数，也就是call方法的第一个参数，可以传null，当this指向null，视为指向window（本文只考虑非严格模式下）。
  2. 函数是可以有返回值的。
  3. context对象中可能存在fn属性。

好在这些问题都是小问题，下面直接贴出最终版的实现：

Function.prototype.myCall = function(context) {
  context = context || window;
  let fn = 0;
  while(context[fn]) {
    fn++;
  }
  context[fn] = this;
  let args = [];
  for (let i = 1; i < arguments.length; i++) {
    args.push('arguments[' + i +']');
  }
  let res = eval('context[fn]('+ args +')');
  delete context[fn];
  return res;
}
// 测试
let obj = {
  a: 1
};
function say(name, age) {
  console.log(name);
  console.log(age);
  console.log(this.a);
  return {
    b: 2
  }
}
console.log(say.myCall(obj, 'Allen', 24));
// Allen
// 24
// 1
// { b: 2}

apply

在MDN中对apply定义如下：

apply() 方法调用一个具有给定this值的函数，以及作为一个数组（或类似数组对象）提供的参数。

注意：call()方法的作用和 apply() 方法类似，区别就是call()方法接受的是参数列表，而apply()方法接受的是一个参数数组。
因为apply的模拟方法与call大致相似，我们这里直接给出代码：

Function.prototype.myApply = function(context) {
  context = context || window;
  let arr = arguments[1] || [];
  let res = null;
  let fn = 0;
  while(context[fn]) {
    fn++;
  }
  context[fn] = this;
  if (arr.length === 0) {
    res = context[fn]();
  } else {
    let args = [];
    for (let i = 0; i < arr.length; i++) {
      args.push('arr[' + i +']');
    }
    res = eval('context[fn]('+ args +')');
  }
  delete context[fn];
  return res;
}
// 按照惯例走波测试
let obj = {
  name: 'Mike'
}
let smart = {
  name: 'Allen',
  getInfo: function(age, sex) {
    console.log(this.name);
    console.log(age);
    console.log(sex);
  }
}
smart.myApply(obj, [24, '男']);
// Mike
// 24
// 男
